{{- /* Returns page headings in the same top-level shape as Page.Fragments.Headings. Hugo may not populate Page.Fragments for external renderers unless the renderer is configured to emit TOC data. When fragments are unavailable, derive a minimal heading tree from rendered HTML so theme TOCs and search anchors still work. @param {Page} . @returns {dict} A "headings" slice with page heading entries. */ -}} {{- $headings := slice -}} {{- with .Fragments -}} {{- with .Headings -}} {{- $headings = . -}} {{- end -}} {{- end -}} {{- if not $headings -}} {{- with .Content -}} {{- $items := slice -}} {{- range findRESubmatch `(?is)]*\bid=["']([^"']+)["'][^>]*>(.*?)` (. | htmlUnescape) -}} {{- $level := int (index . 1) -}} {{- $id := index . 2 -}} {{- $title := index . 3 | plainify | htmlUnescape | strings.TrimSpace -}} {{- if and $id $title -}} {{- $items = $items | append (dict "ID" $id "Level" $level "Title" $title "Headings" slice) -}} {{- end -}} {{- end -}} {{- with $items -}} {{- $headings = slice (dict "ID" "" "Level" 0 "Title" "" "Headings" .) -}} {{- end -}} {{- end -}} {{- end -}} {{- return (dict "headings" $headings) -}}
